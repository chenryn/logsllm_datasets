User: Write a script for batch deployment of applications to multiple servers
Assistant: #!/bin/bash # This script is used to batch deploy applications to multiple servers # Define server list servers=("server1" "server2" "server3") # Define application local path app_path="/path/to/app " # Define remote server application deployment path remote_path="/path/to/remote/app" # Loop deployment application to each server for server in "${servers[@]}" do echo "Deploying application to $server. .." # Copy the application to the remote server scp -r $app_path $server:$remote_path # Start the application on the remote server ssh $server "cd $remote_path; ./startup.sh" echo "Application deployed to $server successfully" done echo "All applications deployed successfully"
