# Examples

## 0.17.1 Print Skull
```vb
Dim I As Integer = 1
Dim a As Integer = 3
Dim b As Integer = 3
Dim c As Integer = 3
Dim d, e, f, g, h, i, j, k, l, m As Integer

Private Sub CreateSkull()
    d = LCDMsgCreate()
    e = LCDMsgCreate()
    f = LCDMsgCreate()
    g = LCDMsgCreate()
    h = LCDMsgCreate()
    i = LCDMsgCreate()
    j = LCDMsgCreate()
    k = LCDMsgCreate()

    LCDMsgSetStr(d, "PWND!----")
    LCDMsgSetStr(e, "1Y1")
    LCDMsgSetStr(f, "11")
    LCDMsgSetStr(g, "1()()1")
    LCDMsgSetStr(h, "1/\1")
    LCDMsgSetStr(i, "11")
    LCDMsgSetStr(j, "LLLLL")
    LCDMsgSetStr(k, "UUUUU")

    LCDMsgMove(d, 10, 30)
    LCDMsgMove(e, 10, 60)
    LCDMsgMove(f, 10, 90)
    LCDMsgMove(g, 10, 120)
    LCDMsgMove(h, 10, 150)
    LCDMsgMove(i, 10, 180)
    LCDMsgMove(j, 10, 210)
    LCDMsgMove(k, 10, 240)
End Sub

Private Sub Initialize()
    SystemCreate()
    CaptureCreate()
    FACreate()
    SSCreate()
    UICreate()
    UICreatePublic()
    DriverCreate()
    StartRedirectUART(1)
    CreateSkull()
    ShowAllTaskInfo()

    For c As Integer = 0 To 10
        For a As Integer = 0 To 10
            LEDDrive(a, 0)
        Next
        BeepDrive(2)
        Wait(500)
        BeepDrive(3)
        For a As Integer = 0 To 10
            LEDDrive(a, 1)
        Next

        ' Colores:
        ' 3 = blanco
        ' 0 = negro
        ' 1 = gris
        ' 5 = verde
        ' >5 = rojo
        LCDMsgChangeColor(d, 0)
        LCDMsgChangeColor(e, 0)
        LCDMsgChangeColor(f, 0)
        LCDMsgChangeColor(g, 0)
        LCDMsgChangeColor(h, 0)
        LCDMsgChangeColor(i, 0)
        LCDMsgChangeColor(j, 0)
        LCDMsgChangeColor(k, 0)
        Wait(500)
        LCDMsgChangeColor(d, 3)
        LCDMsgChangeColor(e, 3)
        LCDMsgChangeColor(f, 3)
        LCDMsgChangeColor(g, 3)
        LCDMsgChangeColor(h, 3)
        LCDMsgChangeColor(i, 3)
        LCDMsgChangeColor(j, 3)
        LCDMsgChangeColor(k, 3)
    Next

    Printf("A\n")
    Printf("A\n")
    Printf("A\n")
    Printf("A\n")
End Sub

Private Sub Terminate()
End Sub
```

## 0.17.2 Record Sound
```vb
Dim I As Integer = 1
Dim a As Integer = 3
Dim b As Integer = 3
Dim c As Integer = 3

Private Sub Initialize()
    SystemCreate()
    CaptureCreate()
    FACreate()
    SSCreate()
    UICreate()
    UICreatePublic()
    StartRedirectUART(1)
    StartFactoryModeController()
    EnableDebugLogMode()
    InitializeSoundRec()
    FreeBufferForSoundRec()
    StartSoundRecord(3)
    TerminateSoundRec()
    ShowTransparentMemory()
    DumpTransparentMemoryItem("RECORDSOUNDDATA")
    ' StartSoundPlay()
    Printf("A\n")
    Printf("A\n")
    Printf("A\n")
    Printf("A\n")
End Sub

Private Sub Terminate()
    a = StrLen("Aaaaaaaa\n")
    Sprint(b, "Aa%caaaaaaa\n", 0)
    a = FopenFut("A/test.txt", "w")
    FwriteFut("AAAA", 1, 4, a)
    FwriteFut(b, 1, 6, a)
    FcloseFut(a)

    For a As Integer = 0 To 10
        Poke8(0xC02200D8, 0x46)
        Wait(100)
        Poke8(0xC02200D8, 0x44)
        Wait(100)
    Next

    Printf("A\n")
    Printf("BBA\n")
    Printf("A\n")
    Printf("A\n")
    ' OpLog.Stop()
End Sub
```

## 0.17.3 Phantom Picture
```vb
Private Sub Say(ByVal b As String)
    Dim a As Integer = LCDMsgCreate()
    LCDMsgSetStr(a, b)
    ShowCameraLog()
End Sub

Private Sub Initialize()
    UICreatePublic()
End Sub

Dim I As Integer = 1
Dim a As Integer = 3
Dim r As Integer = 3
Dim destino As Integer = 0

Private Sub Terminate()
    ' StartRedirectUART(1)
    SystemCreate()
    destino = AllocateMemory(34798)
    r = FopenFut("A/peron.jpg", "rb")
    FreadFut(destino, 34798, 1, r)
    FcloseFut(r)
    a = FopenFut("A/DCIM/100CANON/IMG_3600.JPG", "wb")
    FwriteFut(destino, 34798, 1, a)
    FcloseFut(a)
    Say("Peronist exploit deployed!")
End Sub
```
