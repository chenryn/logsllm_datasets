# 用户侧特征层        user_tower = tf.keras.layers.DenseFeatures(user_feature_columns)(feature_inputs)        
# 连接层及后续多层神经网络        interact_layer = tf.keras.layers.concatenate([item_tower, user_tower])        for num_nodes in hidden_units:            interact_layer = tf.keras.layers.Dense(num_nodes, activation='relu')(interact_layer)
