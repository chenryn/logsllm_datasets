User: Write a scheduled task implemented by Shell script, back up all files in the specified directory every hour and compress them into ZIP format.
Assistant: #!/bin/bash # Define the directory that needs to be backed up backup_dir=/home/user/backup # Define the naming format of the backup file, including date and time backup_file=backup_$(date +%Y%m%d_%H%M) .zip # Perform backup and compression operations zip -r $backup_file $backup_dir # Move the backup file to the specified directory mv $backup_file /home/user/backup_archive echo "Backup succeeded, the backup file is $backup_file"
