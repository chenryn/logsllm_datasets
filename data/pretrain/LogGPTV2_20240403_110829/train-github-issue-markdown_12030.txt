### Example Code
    from fastapi import FastAPI
    from pydantic import BaseModel
    app = FastAPI()
    class MyModel(BaseModel):
        my_int: int
    @app.get("/test", response_model=MyModel)
    async def test():
        model = MyModel(my_int=280456587464933376)
        return model
