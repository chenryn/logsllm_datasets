#!/bin/bash

# 以下为注释示例。
echo "A comment will follow"  # 注释紧随其后
# 注意：在上一条语句中，#号前需要有一个空格。

    # 该注释前有一个制表符。
# 注意：上一条语句中的注释位于行首空白字符之后。

# 在双引号内，# 不会被视为注释开始符。
echo "111-双引号: 这里的 # 并不表示注释。"

# 在单引号内，# 同样不会被视为注释开始符。
echo '222-单引号: 这里的 # 并不表示注释。'

# 当使用反斜杠转义时，# 也不会被视为注释。
echo 333-无引号但已转义: 这里的 \# 并不表示注释。

# 如果没有引号且未转义，# 将作为注释的开始符号。
echo 444-无引号无转义: The # here begins a comment. # 此部分为实际注释

# 在特定参数替换结构中，# 号并不表示注释。
echo ${PATH#*:}  # 功能：删除第一个冒号之前的部分

# 在数字常量表达式中，# 号也不表示注释。
echo $((2#101011))  # 数字进制转换，将二进制数转换为十进制数，这里的 # 不是注释标记
