解密的算法是
    function gw_jiami(text)
        const key="gw"
        dim str : str=key & text 
        dim str1
        dim str2
        for i=1 to len(str)
            str2=hex(AscW(mid(str,i,1)))
            for j=1 to 4-len(str2)
                str2="0" & str2
            next
            str1=str1 & str2
        next
        gw_jiami=strreverse(str1)
    end function
