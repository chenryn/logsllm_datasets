### Skill Permission-to-Policy Consistency Algorithm

**Algorithm 1: Permission-To-Policy Consistency Algorithm**

```python
def PERMTOPOLICYCONSISTENCY(skill, privacyPolicy):
    results = {}
    flows = []
    
    for p in skill.permissions:
        for t in permToDataMap[p]:
            flows.append(FirstPartyCollection(t))
    
    for pcheckRes in PoliCheck(flows, privacyPolicy):
        dataType = pcheckRes.dataType
        for perm in dataToPermMap[dataType]:
            if perm not in results:
                results[perm] = {'stmts': [], 'consistency': None}
            
            results[perm]['stmts'].append(pcheckRes.stmts)
            
            if results[perm]['consistency'] == "PARTIAL":
                if pcheckRes.consistency in ["clear", "vague"]:
                    results[perm]['consistency'] = "PARTIAL"
                    continue
                if pcheckRes.consistency == "CONS":
                    results[perm]['consistency'] = "PARTIAL"
                    continue
                results[perm]['consistency'] = "INCONS"
            elif pcheckRes.consistency == "INCONS":
                results[perm]['consistency'] = "CONS"
            else:
                if not hasPermission(skill, perm):
                    continue
    
    return results
```
