import requests
import zlib
import os

def browser(url, padding):
    # 获取网站内容
    body = requests.get(url, timeout=1).content
    
    # 压缩内容
    compressed = zlib.compress(body)
    
    # 填充到最接近的padding倍数
    final = compressed
    if padding - len(final) > 0:
        final += os.urandom(padding - len(final))
    else:
        final += os.urandom((len(final) + padding) % padding)
